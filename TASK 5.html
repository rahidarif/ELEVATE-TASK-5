import matplotlib.pyplot as plt

# Histogram for Age
plt.figure()
df['Age'].hist()
plt.xlabel('Age')
plt.ylabel('Frequency')
plt.title('Age Distribution')
plt.show()

# Boxplot for Fare
plt.figure()
plt.boxplot(df['Fare'].dropna())

plt.title('Fare Boxplot')
plt.show()
